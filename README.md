    wget -qO- https://raw.githubusercontent.com/byepolr/vim-setup/master/setup.sh | sh

#General Requirements
- CMake
- Nodejs/NPM
- VIM 7.4.1 or greater

##Ubuntu
- sudo apt-get install build-essential cmake
- sudo apt-get install python-dev
